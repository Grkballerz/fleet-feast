```typescript
async function canReview(userId: string, bookingId: string): Promise<boolean> {
  const booking = await prisma.booking.findUnique({
    where: { id: bookingId },
  });

  if (!booking || booking.status !== 'COMPLETED') return false;

  const isParty = booking.customerId === userId || booking.vendorId === userId;
  if (!isParty) return false;

  const existingReview = await prisma.review.findFirst({
    where: { bookingId, reviewerId: userId },
  });

  return !existingReview;
}
```
