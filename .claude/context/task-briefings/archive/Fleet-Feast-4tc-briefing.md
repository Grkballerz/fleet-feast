async function checkLoyaltyEligibility(
  customerId: string,
  vendorId: string
): Promise<LoyaltyStatus> {
  const previousBookings = await prisma.booking.count({
    where: {
      customerId,
      vendorId,
      status: 'COMPLETED',
    },
  });

  return {
    isEligible: previousBookings >= 1,

### Updated Price Calculation
```typescript
const STANDARD_COMMISSION = 0.15;
const LOYALTY_COMMISSION = 0.10;
const LOYALTY_DISCOUNT = 0.05;

function calculateBookingPrice(
  baseAmount: number,
  loyaltyStatus: LoyaltyStatus
) {
  const discountAmount = loyaltyStatus.isEligible
    ? baseAmount * LOYALTY_DISCOUNT
    : 0;

  const totalAmount = baseAmount - discountAmount;

  const commissionRate = loyaltyStatus.isEligible
    ? LOYALTY_COMMISSION
    : STANDARD_COMMISSION;

  const platformFee = totalAmount * commissionRate;
  const vendorPayout = totalAmount - platformFee;

  return {
    baseAmount,
    discountAmount,
    discountPercent: loyaltyStatus.isEligible ? 5 : 0,
    totalAmount,
    platformFee,
    vendorPayout,
    loyaltyApplied: loyaltyStatus.isEligible,
  };
}
```

async function createBooking(data: CreateBookingInput) {
  // Check loyalty eligibility
  const loyaltyStatus = await checkLoyaltyEligibility(
    data.customerId,
    data.vendorId
  );

  // Calculate prices with loyalty
  const pricing = calculateBookingPrice(
    data.baseAmount,
    loyaltyStatus
  );

  const booking = await prisma.booking.create({
    data: {
      ...data,
      totalAmount: pricing.totalAmount,
      discountAmount: pricing.discountAmount,
      loyaltyApplied: pricing.loyaltyApplied,
      platformFee: pricing.platformFee,
      vendorPayout: pricing.vendorPayout,
    },
  });

  return booking;
}
