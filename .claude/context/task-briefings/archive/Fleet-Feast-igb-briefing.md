### Required Schema Addition

model VerificationToken {
  id        String   @id @default(cuid())
  userId    String
  token     String   @unique
  type      TokenType
  expiresAt DateTime
  createdAt DateTime @default(now())

  user      User     @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@unique([userId, type])
  @@index([token])
  @@index([expiresAt])
}

  EMAIL_VERIFICATION
  PASSWORD_RESET
}
